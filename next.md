## Descendent Styles

### On the parent:

```tsx
const ButtonFrame = styled(Stack, {
  ButtonText: {
    color: 'red',
  },

  hoverStyle: {
    ButtonText: {
      color: 'green'
    }
  },

  $small: {
    hoverStyle: {
      ButtonText: {
        color: 'blue'
      }
    }
  }
})

const ButtonText = styled(Stack, {
  name: 'ButtonText',
})
```

### On the child:

```tsx
const ButtonFrame = styled(Stack, {})

const ButtonText = styled(Stack, {
  name: 'ButtonText',

  in_ButtonFrame: {
    color: 'red',

    hoverStyle: {
      color: 'green',
    },

    $small: {
      hoverStyle: {
        color: 'blue'
      }
    }
  },
})
```
