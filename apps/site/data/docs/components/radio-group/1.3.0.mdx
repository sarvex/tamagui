### Usage

```tsx
import { RadioGroup } from 'tamagui'
export default () => (
  <RadioGroup size="$4" value="foo" space="$1">
    <RadioGroup.Item id="dummy" value="foo">
      <RadioGroup.Indicator />
    </RadioGroup.Item>
    <RadioGroup.Item id="other" value="bar">
      <RadioGroup.Indicator />
    </RadioGroup.Item>
  </RadioGroup>
)
```
